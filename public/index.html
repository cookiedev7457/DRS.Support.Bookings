async function renderSessions() { const res = await fetch('/api/sessions'); const sessions = await res.json(); const list = document.getElementById('session-list'); list.innerHTML = sessions.map(s => { const hasJoined = s.participants.some(p => p.userId == user.userId); // Admins see the full list of names; Staff just see the count const participantInfo = user.rank >= 98 ? `Staff Booked: ${s.participants.map(p => p.username).join(', ') || 'None'}` : `Spots Taken: ${s.participants.length}`; return `
${s.sessionName}
Host: ${s.hostName}
${participantInfo}
${user.rank < 98 && !hasJoined ? `` : ''} ${hasJoined ? `✔️ Booked` : ''} ${user.rank >= 98 ? ` ` : ''}
`; }).join('') || "No active sessions."; }
